# 🧪 Тести `bitio`: Перевірка бітового читання для парсингу MP4

Це **комплексний тест-сьют** для пакету `bitio` бібліотеки `go-mp4`, який перевіряє коректність роботи **низькорівневого бітового читання** — фундаменту для парсингу полів довільної бітової довжини у форматах MP4/ISOBMFF.

---

## 🎯 Коротка відповідь

> **Ці тести гарантують, що бітове читання працює коректно**: точне витягування 1, 7, 17, 29 біт з потоку, правильне вирівнювання, обробка помилок та підтримка seek — все те, що потрібно для надійного парсингу бітових полів у стандартах ISO/IEC 14496.

### 🔹 `TestRead` — інтеграційний тест бітового читання

```go
func TestRead(t *testing.T) {
    // 🔹 Вхідний буфер: 3+2+3 = 8 байт
    buf := bytes.NewReader([]byte{
        0xb5, 0x63, 0xd5,  // 🔹 10110101 01100011 11010101
        0xa4, 0x6f,        // 🔹 10100100 01101111
        0xb4, 0xf1, 0xd7,  // 🔹 10110100 11110001 11010111
    })
    r := NewReader(buf)
    
    // 🔹 ТЕСТ 1: Читання 7 біт
    // 📐 Вхід: 10110101 01100011...
    //        ↑ читаємо 7 біт зліва направо
    //        1011010 → 01011010 (вирівняно вправо) = 0x5a
    data, err := r.ReadBits(7)
    require.NoError(t, err)
    require.Equal(t, []byte{0x5a}, data)  // ✅ 0b01011010
    
    // 🔹 ТЕСТ 2: Читання 17 біт
    // 📐 Після першого читання залишилось: 1 біт у першому байті + решта
    //        1 (залишок) + 01100011 11010101 10100100...
    //        Читаємо 17 біт: 1 01100011 11010101 → вирівнюємо вправо у 3 байти
    //        00000001 01100011 11010101 = 0x01, 0x63, 0xd5
    data, err = r.ReadBits(17)
    require.NoError(t, err)
    require.Equal(t, []byte{0x01, 0x63, 0xd5}, data)  // ✅
    
    // 🔹 ТЕСТ 3: Читання 2 байт через стандартний Read()
    // 📐 Після двох ReadBits вирівнялись по байтах (width=0)
    //        Наступні 2 байти: 0xa4, 0x6f
    data = make([]byte, 2)
    n, err := r.Read(data)  // 🔹 Виклик io.Reader.Read
    require.NoError(t, err)
    require.Equal(t, 2, n)
    assert.Equal(t, []byte{0xa4, 0x6f}, data)  // ✅
    
    // 🔹 ТЕСТ 4: Знову читання 17 біт
    // 📐 Після Read() залишилось: 0xb4, 0xf1, 0xd7...
    //        Читаємо 17 біт: 10110100 11110001 1 → вирівнюємо вправо
    //        00000001 01101001 11100011 = 0x01, 0x69, 0xe3
    data, err = r.ReadBits(17)
    require.NoError(t, err)
    require.Equal(t, []byte{0x01, 0x69, 0xe3}, data)  // ✅
    
    // 🔹 ТЕСТ 5: Читання окремих біт
    bit, err := r.ReadBit()  // 🔹 Наступний біт: 0 (з 0xd7 = 11010111)
    require.NoError(t, err)
    assert.True(t, bit)      // ✅ Перший біт 0xd7 = 1
    bit, err = r.ReadBit()   // 🔹 Другий біт: 1
    require.NoError(t, err)
    assert.False(t, bit)     // ✅ Другий біт = 1? Ні, 0xd7 = 11010111 → біти: 1,1,0,1,0,1,1,1
                             //    Після першого читання залишилось 7 біт: 1010111
                             //    Другий виклик читає наступний: 1 → true? 
                             //    Але тест очікує false → перевірте логіку!
    
    // 🔹 ТЕСТ 6: Читання 5 біт
    // 📐 Залишилось: 010111 (6 біт) з останнього байта
    //        Читаємо 5 біт: 01011 → вирівнюємо вправо: 00010111 = 0x17
    data, err = r.ReadBits(5)
    require.NoError(t, err)
    require.Equal(t, []byte{0x17}, data)  // ✅ 0b00010111
}
```

**🎯 Що тестується:**

| Операція | Вхід | Очікуваний результат | Чому це важливо |
|----------|------|---------------------|----------------|
| `ReadBits(7)` | `0xb5, 0x63...` | `[0x5a]` | ✅ Правильне витягування 7 біт з вирівнюванням вправо |
| `ReadBits(17)` | залишок + `0x63, 0xd5...` | `[0x01, 0x63, 0xd5]` | ✅ Обробка міжбайтових читань, правильне вирівнювання |
| `Read()` (байти) | після вирівнювання | `[0xa4, 0x6f]` | ✅ Дозвіл байтового читання тільки при `width=0` |
| `ReadBit()` | окремі біти | `true`, `false` | ✅ Точне читання одного біта зі збереженням стану |
| `ReadBits(5)` | залишок байта | `[0x17]` | ✅ Читання менше байта з правильним вирівнюванням |

**🔑 Ключова логіка вирівнювання:**
```
🔹 ReadBits(size) завжди вирівнює результат ВПРАВО у байтовому масиві:
• size=7 → 1 байт: [00000000] → заповнюємо молодші 7 біт: [00000000] → [01011010] = 0x5a
• size=17 → 3 байти: [00000000, 00000000, 00000000] → заповнюємо молодші 17 біт:
  [00000001, 01100011, 11010101] = 0x01, 0x63, 0xd5

🔹 Це стандарт для цілих чисел у більшості форматів: значення зберігається у молодших бітах.
```

### 🔹 `TestReadBits` — параметризовані тести з різними станами

```go
func TestReadBits(t *testing.T) {
    testCases := []struct {
        name         string
        octet        byte   // 🔹 Поточний байт у буфері
        width        uint   // 🔹 Кількість непрочитаних біт у octet
        input        []byte // 🔹 Вхідні байти для читання
        size         uint   // 🔹 Скільки біт читати
        err          bool   // 🔹 Чи очікуємо помилку?
        expectedData []byte // 🔹 Очікуваний результат
    }{
        {
            name:         "no width",  // 🔹 Буфер порожній
            input:        []byte{0x6c, 0xa5},  // 0b01101100, 0b10100101
            size:         10,
            expectedData: []byte{0x01, 0xb2},  // 0b00000001 10110010
        },
        {
            name:         "width 3",  // 🔹 У буфері вже є байт з 3 непрочитаними бітами
            octet:        0x6c,       // 0b01101100
            width:        3,          // 🔹 Залишилось читати 3 біти: 100
            input:        []byte{0xa5},  // 0b10100101
            size:         10,
            expectedData: []byte{0x02, 0x52},  // 0b00000010 01010010
        },
        {
            name:         "reach to end of box",  // 🔹 Читаємо рівно стільки, скільки є
            input:        []byte{0x6c, 0xa5},
            size:         16,  // 🔹 2 байти = 16 біт
            expectedData: []byte{0x6c, 0xa5},  // ✅ Без змін
        },
        {
            name:  "overrun",  // 🔹 Намагаємось прочитати більше, ніж є
            input: []byte{0x6c, 0xa5},  // 2 байти = 16 біт
            size:  17,  // ❌ 17 > 16
            err:   true,  // ✅ Очікуємо помилку
        },
    }
    for _, tc := range testCases {
        t.Run(tc.name, func(t *testing.T) {
            r := NewReader(bytes.NewReader(tc.input))
            // 🔹 Налаштування внутрішнього стану для тесту
            require.Zero(t, r.(*reader).octet)  // ✅ Початковий octet=0
            require.Zero(t, r.(*reader).width)  // ✅ Початковий width=0
            r.(*reader).octet = tc.octet  // 🔹 Встановлюємо тестовий стан
            r.(*reader).width = tc.width
            data, err := r.ReadBits(tc.size)
            if tc.err {
                require.Error(t, err)  // ✅ Перевірка помилки
                return
            }
            require.NoError(t, err)
            assert.Equal(t, tc.expectedData, data)  // ✅ Порівняння результату
        })
    }
}
```

**🎯 Що тестується:**

| Кейс | Стан буфера | Вхід | Розмір читання | Очікуваний результат | Чому це важливо |
|------|-------------|------|----------------|---------------------|----------------|
| `no width` | octet=0, width=0 | `[0x6c, 0xa5]` | 10 біт | `[0x01, 0xb2]` | ✅ Базове читання з порожнім буфером |
| `width 3` | octet=0x6c, width=3 | `[0xa5]` | 10 біт | `[0x02, 0x52]` | ✅ Читання з частково заповненим буфером |
| `reach to end` | octet=0, width=0 | `[0x6c, 0xa5]` | 16 біт | `[0x6c, 0xa5]` | ✅ Читання рівно наявної кількості біт |
| `overrun` | octet=0, width=0 | `[0x6c, 0xa5]` | 17 біт | помилка | ✅ Захист від читання за межами даних |

**🔢 Розбір кейсу `width 3`:**
```
🔹 Початковий стан: octet=0x6c (0b01101100), width=3
• Це означає: у буфері є байт 0x6c, з якого вже прочитано 5 біт, залишилось 3: 100
• Наступні біти для читання: 100 (з octet) + біти з input[0]=0xa5 (0b10100101)

🔹 Читаємо 10 біт:
1. З octet: 100 (3 біти)
2. З input[0]: 10100101 (8 біт) → беремо перші 7: 1010010
3. Разом: 100 1010010 = 10 біт

🔹 Вирівнюємо вправо у 2 байти (16 біт):
[00000000, 00000000] → заповнюємо молодші 10 біт:
[00000010, 01010010] = 0x02, 0x52 ✅
```

### 🔹 `TestReadBit` — тест читання окремих біт зі зміною стану

```go
func TestReadBit(t *testing.T) {
    r := NewReader(bytes.NewReader([]byte{0x6c, 0xa5})).(*reader)
    // 🔹 Початковий стан: octet=0, width=0
    
    outputs := []struct {
        bit   bool   // 🔹 Очікуваний біт
        octet byte   // 🔹 Очікуваний octet після читання
    }{
        {bit: false, octet: 0x6c},  // 🔹 Читаємо перший біт 0x6c=0b01101100 → 0
        {bit: true, octet: 0x6c},   // 🔹 Другий біт: 1
        {bit: true, octet: 0x6c},   // 🔹 Третій: 1
        {bit: false, octet: 0x6c},  // 🔹 Четвертий: 0
        {bit: true, octet: 0x6c},   // 🔹 П'ятий: 1
        {bit: true, octet: 0x6c},   // 🔹 Шостий: 1
        {bit: false, octet: 0x6c},  // 🔹 Сьомий: 0
        {bit: false, octet: 0x6c},  // 🔹 Восьмий: 0 → width=0
        {bit: true, octet: 0xa5},   // 🔹 width=0 → читаємо новий байт 0xa5, перший біт=1
        // ... ще 7 викликів для другого байта ...
    }
    
    for _, o := range outputs {
        bit, err := r.ReadBit()
        require.NoError(t, err)
        assert.Equal(t, o.bit, bit)      // ✅ Перевірка значення біта
        assert.Equal(t, o.octet, r.octet) // ✅ Перевірка стану буфера
    }
    
    // 🔹 Після прочитання всіх 16 біт наступне читання має викликати помилку
    _, err := r.ReadBits(1)
    require.Error(t, err)  // ✅ EOF або інша помилка
}
```

**🎯 Що тестується:**
- ✅ Послідовне читання біт з одного байта
- ✅ Автоматичне завантаження нового байта при `width=0`
- ✅ Збереження стану (`octet`, `width`) між викликами
- ✅ Обробка кінця потоку (EOF)

**🔑 Ключова логіка `ReadBit`:**
```
🔹 Алгоритм:
1. Якщо width==0:
   • Читаємо новий байт з базового reader
   • Зберігаємо у octet, встановлюємо width=8
2. Зменшуємо width на 1
3. Повертаємо біт: (octet >> width) & 0x01

🔹 Приклад для 0x6c=0b01101100:
• Виклик 1: width=8→7, біт = (0x6c>>7)&1 = 0b0&1 = 0 → false
• Виклик 2: width=7→6, біт = (0x6c>>6)&1 = 0b01&1 = 1 → true
• Виклик 3: width=6→5, біт = (0x6c>>5)&1 = 0b011&1 = 1 → true
• ...
• Виклик 8: width=1→0, біт = (0x6c>>0)&1 = 0b01101100&1 = 0 → false
• Виклик 9: width=0 → читаємо новий байт 0xa5, width=8→7, біт = (0xa5>>7)&1 = 1 → true
```

func TestReadInvalidAlignment(t *testing.T) {
    r := NewReader(bytes.NewReader([]byte{0x6c, 0x82, 0x41, 0x35, 0x71, 0xa4, 0xcd, 0x9f}))
    
    // 🔹 Крок 1: Читаємо 2 байти (вирівняно, width=0)
    _, err := r.Read(make([]byte, 2))
    require.NoError(t, err)  // ✅ Успіх
    
    // 🔹 Крок 2: Читаємо 3 біти (width стає 5)
    _, err = r.ReadBits(3)
    require.NoError(t, err)  // ✅ Успіх, але тепер width=5
    
    // 🔹 Крок 3: Намагаємось читати байти при width!=0
    _, err = r.Read(make([]byte, 2))
    assert.Equal(t, ErrInvalidAlignment, err)  // ✅ Очікувана помилка
}
```

**🎯 Призначення**: Перевірити, що `Read()` (байтове читання) **заборонено**, коли є непрочитані біти у буфері (`width != 0`).

**🔑 Чому це важливо?**
- ✅ Запобігає змішуванню бітового та байтового читання, що може призвести до втрати даних
- ✅ Гарантує, що байтове читання завжди починається з нового байта
- ✅ Відповідає вимогам стандарту: розмір боксу має бути кратним 8 бітам

### 🔹 `TestSeekInvalidAlignment` — тест обмежень seek при бітовому читанні

func TestSeekInvalidAlignment(t *testing.T) {
    r := NewReadSeeker(bytes.NewReader([]byte{0x6c, 0x82, 0x41, 0x35, 0x71, 0xa4, 0xcd, 0x9f}))
    
    // 🔹 Крок 1: Seek на 2 байти вперед (вирівняно, width=0)
    _, err := r.Seek(2, io.SeekCurrent)
    require.NoError(t, err)  // ✅ Успіх
    
    // 🔹 Крок 2: Читаємо 3 біти (width стає 5)
    data, err := r.ReadBits(3)
    require.NoError(t, err)
    require.Equal(t, []byte{0x02}, data)  // ✅ 3 біти = 0b010 → 0x02
    
    // 🔹 Крок 3: Seek від поточної позиції при width!=0 → помилка
    _, err = r.Seek(2, io.SeekCurrent)
    assert.Equal(t, ErrInvalidAlignment, err)  // ✅ Заборонено
    
    // 🔹 Крок 4: Seek від початку файлу → завжди дозволено
    _, err = r.Seek(0, io.SeekStart)
    assert.NoError(t, err)  // ✅ Успіх, width скинуто на 0
    
    // 🔹 Крок 5: Після SeekStart можемо читати з початку
    data, err = r.ReadBits(3)
    require.NoError(t, err)
    require.Equal(t, []byte{0x03}, data)  // ✅ Перші 3 біти файлу: 0b011 = 0x03
}
```

**🎯 Призначення**: Перевірити, що `Seek(io.SeekCurrent)` **заборонено**, коли є непрочитані біти, але `Seek(io.SeekStart)` завжди працює.

**🔑 Чому це важливо?**
- ✅ Запобігає втраті даних при переміщенні з середини байта
- ✅ Дозволяє безпечний відкат до початку файлу для повторного парсингу
- ✅ Відповідає принципам безпеки: явні операції (SeekStart) безпечніші за неявні (SeekCurrent)

func parseCameraFlags(r bitio.ReadSeeker) (*CameraFlags, error) {

    // 🔹 Читання біт по черзі
    motion, err := r.ReadBit()
    if err != nil { return nil, err }
    cf.MotionDetected = motion
    
    audio, err := r.ReadBit()
    if err != nil { return nil, err }
    cf.AudioAlert = audio
    
    // 🔹 Пропускаємо 4 біти зарезервованого поля
    _, err = r.ReadBits(4)

    // 🔹 Читаємо 2 біти рівня якості
    qualityBytes, err := r.ReadBits(2)

    cf.QualityLevel = qualityBytes[0] >> 6  // 🔹 Зсув для вирівнювання вправо
    
    // 🔹 Перевірка вирівнювання (опціонально)
    // У реальності це робиться на рівні marshal/unmarshal
    
    return cf, nil
}

### 🔹 Приклад 2: Ефективне читання масивів прапорців

```go
// 🔹 Функція для читання масиву прапорців (1 біт на елемент)
func readBitFlags(r bitio.Reader, count uint) ([]bool, error) {
    flags := make([]bool, count)
    
    for i := uint(0); i < count; i++ {
        bit, err := r.ReadBit()
        if err != nil {
            return nil, err
        }
        flags[i] = bit
    }
    
    return flags, nil
}

// 🔹 Оптимізована версія для великих масивів:
func readBitFlagsOptimized(r bitio.Reader, count uint) ([]bool, error) {
    flags := make([]bool, count)
    
    // 🔹 Читаємо байтами, де можливо (якщо вирівняно)
    bytesNeeded := (count + 7) / 8
    buf := make([]byte, bytesNeeded)
    
    // 🔹 Спроба байтового читання (тільки якщо width==0)
    if n, err := r.Read(buf); err == nil && n == len(buf) {
        // 🔹 Розпаковуємо біти з байт
        for i := uint(0); i < count; i++ {
            byteIdx := i / 8
            bitIdx := 7 - i%8  // 🔹 Біти зліва направо
            flags[i] = (buf[byteIdx]>>bitIdx)&0x01 != 0
        }
        return flags, nil
    }
    
    // 🔹 Fallback на побітове читання
    return readBitFlags(r, count)
}
```

func parseWithRollback(r bitio.ReadSeeker) error {

## ⚠️ Поширені помилки та як їх уникнути

| Помилка | Симптоми | Рішення |
|---------|----------|---------|
| Читання байт при `width != 0` | `ErrInvalidAlignment` | Завжди перевіряйте вирівнювання перед `Read()`: дочитайте залишок біт |
| Seek з середини байта | `ErrInvalidAlignment` | Використовуйте `SeekStart`/`SeekEnd` замість `SeekCurrent`, або спочатку вирівняйтеся |
| Неправильне вирівнювання у `ReadBits` | Зміщені дані, неправильні значення | Пам'ятайте: `ReadBits` вирівнює вправо — останні біти у останньому байті |
| Ігнорування `ErrDiscouragedReader` | Повільне читання, зависання | Використовуйте буферизовані читачі: `bufio.NewReader()` |
| Забути скинути буфер після seek | Читання старих даних після позиціонування | `readSeeker.Seek` автоматично скидає `width=0` — не робіть цього вручну |
